384

387  //排除系统目录下的dll

393  //排除exe文件
394  return false;

396

399 static bool dllcopy(std::string dll)
400 {
401  if(dll.empty())
402  {
403  return false;
404  }
405  std::string processdir=GetCurrentProcessDir();
406  if(processdir.at(processdir.length()-1)!='\\')
407  {
408  processdir+='\\';
409  }
410  std::string dllname;
411  {
412  //获取dll名称
413  for(size_t i=0; i<dll.length(); i++)
414  {
415  if(dll.at(dll.length()-i-1)=='\\')
416  {
417  dllname=dll.substr(dll.length()-i);
418  break;
419  }
420  }
421  if(dllname.empty())
422  {
423  dllname=dll;
424  }
425  }
426  std::string newdll=(processdir+dllname);
427  return 0!=CopyFileA(dll.c_str(),newdll.c_str(),0);
428 }

430 static void deploywin_thread(deploywin_startup * obj)
431 {
432  while(true)
433  {
434  while(!dllqueueisempty())
435  {
436  IsCopying=true;
437  std::string dll=dllqueueget();
438  if(dllneedcopy(dll))
439  {
440  printf("DLL(%s) need copy!\n",dll.c_str());
441  if(dllcopy(dll))
442  {
443  printf("DLL(%s) copy success!\n",dll.c_str());
444  }
445  }
446  IsCopying=false;
447 
448  }
449  {
450  //自动退出
451  if(GetAutoExit()>0)
452  {
453  if(std::chrono::steady_clock::now()-obj->GetStartTime() > std::chrono::seconds(GetAutoExit()))
454  {
455  printf("DeployWin will exit!\n");
456  exit(0);
457  }
458  }
459  }
460  std::this_thread::sleep_for(std::chrono::seconds(1));
461  }
